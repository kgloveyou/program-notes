# 积累

1、文字前添加蓝色方块。

```tsx
import React from 'react';
import trans from '@/utils/translate';

interface Props {
  title: string,
}
const Title: React.FC<Props> = (props) => {
  return (
    <div style={{ marginBottom: '20px', verticalAlign: 'center' }}>
      <span
        style={{
          display: 'inline-block',
          width: '2px',
          height: '12px',
          backgroundColor: 'blue',
          marginRight: '10px',
        }}
      ></span>
      {trans(props.title)}
    </div>
  )
}

export default Title;
```
